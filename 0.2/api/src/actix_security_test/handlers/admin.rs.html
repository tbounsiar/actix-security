1//! Admin routes (protected by middleware via RequestMatcherAuthorizer).
2
3use actix_web::{get, HttpResponse, Responder};
4
5use actix_security::http::security::AuthenticatedUser;
6
7/// Admin-only endpoint (requires ADMIN role - protected by middleware).
8#[get("/admin/dashboard")]
9pub async fn admin_dashboard(user: AuthenticatedUser) -> impl Responder {
10    HttpResponse::Ok().body(format!(
11        "Admin Dashboard\n\nWelcome, {}!\nYou have admin access.",
12        user.get_username()
13    ))
14}
15
16/// Admin users management (protected by middleware).
17#[get("/admin/users")]
18pub async fn admin_users(user: AuthenticatedUser) -> impl Responder {
19    HttpResponse::Ok().body(format!(
20        "User Management (Admin)\n\nLogged in as: {}\nThis page is only for ADMIN role.",
21        user.get_username()
22    ))
23}
24
25/// User area (requires USER or ADMIN role - protected by middleware).
26#[get("/user/settings")]
27pub async fn user_settings(user: AuthenticatedUser) -> impl Responder {
28    HttpResponse::Ok().body(format!(
29        "User Settings\n\nUser: {}\nRoles: {:?}",
30        user.get_username(),
31        user.get_roles()
32    ))
33}
