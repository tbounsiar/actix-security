1//! Routes protected by #[secured] macro.
2//!
3//! # Spring Security Equivalent
4//! `@Secured` annotation
5
6use actix_web::{get, HttpResponse, Responder};
7
8use actix_security::http::security::AuthenticatedUser;
9use actix_security::secured;
10
11/// Reports page - protected by #[secured] macro (ADMIN role required).
12///
13/// # Spring Security Equivalent
14/// `@Secured("ROLE_ADMIN")`
15#[secured("ADMIN")]
16#[get("/reports")]
17pub async fn reports(user: AuthenticatedUser) -> impl Responder {
18    HttpResponse::Ok().body(format!(
19        "Reports Dashboard\n\nGenerated for: {}\nThis endpoint uses #[secured(\"ADMIN\")] macro.",
20        user.get_username()
21    ))
22}
23
24/// Management page - protected by #[secured] macro (ADMIN or MANAGER role).
25///
26/// # Spring Security Equivalent
27/// `@Secured({"ROLE_ADMIN", "ROLE_MANAGER"})`
28#[secured("ADMIN", "MANAGER")]
29#[get("/management")]
30pub async fn management(user: AuthenticatedUser) -> impl Responder {
31    HttpResponse::Ok().body(format!(
32        "Management Panel\n\nUser: {}\nRoles: {:?}\nThis endpoint requires ADMIN or MANAGER role.",
33        user.get_username(),
34        user.get_roles()
35    ))
36}
