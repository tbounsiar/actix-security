1//! HTTP Basic Authentication Example
2//!
3//! This example demonstrates HTTP Basic authentication with Actix Security.
4//!
5//! # Run
6//! ```bash
7//! cargo run --example basic_auth
8//! ```
9//!
10//! # Test
11//! ```bash
12//! # Successful authentication
13//! curl -u admin:admin http://localhost:8080/
14//! curl -u user:user http://localhost:8080/
15//!
16//! # Access admin-only route
17//! curl -u admin:admin http://localhost:8080/admin
18//! curl -u user:user http://localhost:8080/admin  # 403 Forbidden
19//!
20//! # Without authentication
21//! curl http://localhost:8080/  # 401 Unauthorized
22//! ```
23
24use actix_security::http::security::web::{Access, MemoryAuthenticator, RequestMatcherAuthorizer};
25use actix_security::http::security::{
26    middleware::SecurityTransform, Argon2PasswordEncoder, AuthenticatedUser, AuthenticationManager,
27    AuthorizationManager, PasswordEncoder, User,
28};
29use actix_web::{get, App, HttpResponse, HttpServer, Responder};
30
31/// Creates the authenticator with users stored in memory.
32fn authenticator() -> MemoryAuthenticator {
33    let encoder = Argon2PasswordEncoder::new();
34
35    AuthenticationManager::in_memory_authentication()
36        .password_encoder(encoder.clone())
37        .with_user(
38            User::with_encoded_password("admin", encoder.encode("admin"))
39                .roles(&["ADMIN".into(), "USER".into()])
40                .authorities(&["users:read".into(), "users:write".into()]),
41        )
42        .with_user(
43            User::with_encoded_password("user", encoder.encode("user"))
44                .roles(&["USER".into()])
45                .authorities(&["users:read".into()]),
46        )
47}
48
49/// Creates the authorizer with URL-based access control.
50fn authorizer() -> RequestMatcherAuthorizer {
51    AuthorizationManager::request_matcher()
52        .http_basic() // Enable HTTP Basic Authentication
53        .add_matcher("/admin.*", Access::new().roles(vec!["ADMIN"]))
54        .add_matcher("/api/.*", Access::new().authorities(vec!["users:read"]))
55}
56
57#[get("/")]
58async fn index(user: AuthenticatedUser) -> impl Responder {
59    HttpResponse::Ok().body(format!(
60        "Hello, {}! Your roles: {:?}",
61        user.get_username(),
62        user.get_roles()
63    ))
64}
65
66#[get("/admin")]
67async fn admin(user: AuthenticatedUser) -> impl Responder {
68    HttpResponse::Ok().body(format!("Welcome to admin panel, {}!", user.get_username()))
69}
70
71#[get("/api/users")]
72async fn api_users(user: AuthenticatedUser) -> impl Responder {
73    HttpResponse::Ok().json(serde_json::json!({
74        "users": ["alice", "bob", "charlie"],
75        "requested_by": user.get_username()
76    }))
77}
78
79#[actix_web::main]
80async fn main() -> std::io::Result<()> {
81    env_logger::init_from_env(env_logger::Env::default().default_filter_or("info"));
82
83    println!("=== HTTP Basic Authentication Example ===");
84    println!("Server running at http://localhost:8080");
85    println!();
86    println!("Test users:");
87    println!("  admin:admin - Roles: [ADMIN, USER]");
88    println!("  user:user   - Roles: [USER]");
89    println!();
90    println!("Try:");
91    println!("  curl -u admin:admin http://localhost:8080/");
92    println!("  curl -u admin:admin http://localhost:8080/admin");
93    println!("  curl -u user:user http://localhost:8080/admin  # 403");
94    println!();
95
96    HttpServer::new(move || {
97        App::new()
98            .wrap(
99                SecurityTransform::new()
100                    .config_authenticator(authenticator)
101                    .config_authorizer(authorizer),
102            )
103            .service(index)
104            .service(admin)
105            .service(api_users)
106    })
107    .bind("127.0.0.1:8080")?
108    .run()
109    .await
110}
