1//! Custom Expressions with Parameter References Example
2//!
3//! This example demonstrates Spring Security-style custom expressions
4//! with parameter references (`#param` syntax) in `#[pre_authorize]`.
5//!
6//! # Run
7//! ```bash
8//! cargo run --bin custom_expressions
9//! ```
10//!
11//! # Test
12//! ```bash
13//! # Tenant access - admin can access any tenant
14//! curl -u admin:admin http://localhost:8080/tenants/123
15//!
16//! # Tenant access - user can only access their own tenant (1)
17//! curl -u user:user http://localhost:8080/tenants/1       # OK
18//! curl -u user:user http://localhost:8080/tenants/2       # 403 Forbidden
19//!
20//! # Resource access - admin can access any resource
21//! curl -u admin:admin http://localhost:8080/resources/private-doc
22//!
23//! # Resource access - user can only access public resources
24//! curl -u user:user http://localhost:8080/resources/public-doc   # OK
25//! curl -u user:user http://localhost:8080/resources/private-doc  # 403 Forbidden
26//!
27//! # Combined expression - hasRole('ADMIN') OR is_tenant_admin(#tenant_id)
28//! curl -u admin:admin http://localhost:8080/tenants/999/settings  # OK (admin role)
29//! curl -u user:user http://localhost:8080/tenants/1/settings      # OK (tenant admin)
30//! curl -u user:user http://localhost:8080/tenants/2/settings      # 403 Forbidden
31//!
32//! # Query parameter - search with min price filter
33//! curl -u user:user "http://localhost:8080/products/search?min_price=50"   # OK
34//! curl -u user:user "http://localhost:8080/products/search?min_price=500"  # 403 (premium only)
35//!
36//! # JSON body - create order with amount validation
37//! curl -u user:user -X POST http://localhost:8080/orders \
38//!      -H "Content-Type: application/json" \
39//!      -d '{"amount": 100}'   # OK
40//! curl -u user:user -X POST http://localhost:8080/orders \
41//!      -H "Content-Type: application/json" \
42//!      -d '{"amount": 5000}'  # 403 (requires ADMIN for > 1000)
43//! ```

59/// Custom authorization function: checks if user is admin of a specific tenant.
60///
61/// This function demonstrates how to create custom authorization logic
62/// that references handler parameters.
63pub async fn is_tenant_admin(user: &User, tenant_id: i64) -> bool {
64    // Admin user can access all tenants
65    if user.has_role("ADMIN") {
66        return true;
67    }
68    // User "user" is admin of tenant 1 only
69    if user.get_username() == "user" && tenant_id == 1 {
70        return true;
71    }
72    false
73}
74
75/// Custom authorization function: checks if user can access a specific resource.
76///
77/// Demonstrates string parameter handling.
78pub async fn can_access_resource(user: &User, resource_id: String) -> bool {
79    // Admin can access all resources
80    if user.has_role("ADMIN") {
81        return true;
82    }
83    // Regular users can only access public resources
84    resource_id.starts_with("public-")
85}
86
87/// Custom authorization function: checks if user can perform premium search.
88///
89/// Demonstrates Query parameter validation. Accepts the full SearchQuery struct.
90pub async fn can_search_premium(user: &User, query: SearchQuery) -> bool {
91    // Admin and premium users can search with any price filter
92    if user.has_role("ADMIN") || user.has_authority("premium:access") {
93        return true;
94    }
95    // Regular users limited to min_price <= 100
96    query.min_price <= 100
97}
98
99/// Custom authorization function: validates order amount based on user role.
100///
101/// Demonstrates Json body parameter validation. Accepts the full CreateOrderRequest struct.
102pub async fn can_create_order(user: &User, order: CreateOrderRequest) -> bool {
103    // Admin can create orders of any amount
104    if user.has_role("ADMIN") {
105        return true;
106    }
107    // Regular users limited to orders <= 1000
108    order.amount <= 1000
109}

129/// Creates the authenticator with users stored in memory.
130fn authenticator() -> MemoryAuthenticator {
131    let encoder = Argon2PasswordEncoder::new();
132
133    AuthenticationManager::in_memory_authentication()
134        .password_encoder(encoder.clone())
135        .with_user(
136            User::with_encoded_password("admin", encoder.encode("admin"))
137                .roles(&["ADMIN".into(), "USER".into()])
138                .authorities(&["users:read".into(), "users:write".into()]),
139        )
140        .with_user(
141            User::with_encoded_password("user", encoder.encode("user"))
142                .roles(&["USER".into()])
143                .authorities(&["users:read".into()]),
144        )
145        .with_user(
146            User::with_encoded_password("premium", encoder.encode("premium"))
147                .roles(&["USER".into()])
148                .authorities(&["users:read".into(), "premium:access".into()]),
149        )
150        .with_user(
151            User::with_encoded_password("guest", encoder.encode("guest"))
152                .roles(&["GUEST".into()]),
153        )
154}
155
156/// Creates the authorizer with HTTP Basic auth enabled.
157fn authorizer() -> RequestMatcherAuthorizer {
158    AuthorizationManager::request_matcher().http_basic()
159}

165/// Handler using custom function with Path parameter.
166///
167/// The `#tenant_id` references the `tenant_id` path parameter.
168#[pre_authorize("is_tenant_admin(#tenant_id)")]
169#[get("/tenants/{tenant_id}")]
170pub async fn get_tenant(tenant_id: web::Path<i64>, user: AuthenticatedUser) -> impl Responder {
171    HttpResponse::Ok().body(format!(
172        "Tenant {} accessed by {}",
173        tenant_id.into_inner(),
174        user.get_username()
175    ))
176}
177
178/// Handler combining custom function with built-in expression.
179///
180/// Access is granted if user hasRole('ADMIN') OR is_tenant_admin of the tenant.
181#[pre_authorize("hasRole('ADMIN') OR is_tenant_admin(#tenant_id)")]
182#[get("/tenants/{tenant_id}/settings")]
183pub async fn get_tenant_settings(
184    tenant_id: web::Path<i64>,
185    user: AuthenticatedUser,
186) -> impl Responder {
187    HttpResponse::Ok().body(format!(
188        "Tenant {} settings for {}",
189        tenant_id.into_inner(),
190        user.get_username()
191    ))
192}
193
194/// Handler with string Path parameter.
195#[pre_authorize("can_access_resource(#resource_id)")]
196#[get("/resources/{resource_id}")]
197pub async fn get_resource(
198    resource_id: web::Path<String>,
199    user: AuthenticatedUser,
200) -> impl Responder {
201    HttpResponse::Ok().body(format!(
202        "Resource '{}' accessed by {}",
203        resource_id.into_inner(),
204        user.get_username()
205    ))
206}

208/// Handler with Query parameter validation.
209///
210/// The `#query` references the Query<SearchQuery> extractor parameter.
211#[pre_authorize("can_search_premium(#query)")]
212#[get("/products/search")]
213pub async fn search_products(
214    query: web::Query<SearchQuery>,
215    user: AuthenticatedUser,
216) -> impl Responder {
217    HttpResponse::Ok().json(serde_json::json!({
218        "query": {
219            "min_price": query.min_price
220        },
221        "results": ["Product A", "Product B"],
222        "searched_by": user.get_username()
223    }))
224}
225
226/// Handler with Json body parameter validation.
227///
228/// The `#body` references the Json<CreateOrderRequest> extractor parameter.
229#[pre_authorize("can_create_order(#body)")]
230#[post("/orders")]
231pub async fn create_order(
232    body: web::Json<CreateOrderRequest>,
233    user: AuthenticatedUser,
234) -> impl Responder {
235    HttpResponse::Created().json(serde_json::json!({
236        "order_id": "ORD-12345",
237        "amount": body.amount,
238        "created_by": user.get_username()
239    }))
240}
241
242/// Handler with multiple conditions on body fields.
243#[pre_authorize("hasRole('ADMIN') OR can_create_order(#body)")]
244#[post("/orders/bulk")]
245pub async fn create_bulk_orders(
246    body: web::Json<CreateOrderRequest>,
247    user: AuthenticatedUser,
248) -> impl Responder {
249    HttpResponse::Created().json(serde_json::json!({
250        "message": "Bulk order created",
251        "amount": body.amount,
252        "created_by": user.get_username()
253    }))
254}

293        App::new()
294            .wrap(

302            .service(search_products)
