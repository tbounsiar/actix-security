18/// Parsed arguments for #[pre_authorize(...)]
19pub enum PreAuthorizeCheck {
20    /// #[pre_authorize(authenticated)]
21    Authenticated,
22    /// #[pre_authorize(role = "ADMIN")] or #[pre_authorize(roles = ["ADMIN", "USER"])]
23    Roles(Vec<String>),
24    /// #[pre_authorize(authority = "read")] or #[pre_authorize(authorities = ["read", "write"])]
25    Authorities(Vec<String>),
26    /// #[pre_authorize("hasRole('ADMIN') OR hasAuthority('write')")]
27    Expression(String),
28}
29
30pub struct PreAuthorizeArgs {
31    pub check: PreAuthorizeCheck,
32}
33
34impl Parse for PreAuthorizeArgs {
35    fn parse(input: ParseStream) -> syn::Result<Self> {
36        // Check for string literal (expression syntax)
37        if input.peek(syn::LitStr) {
38            let lit: syn::LitStr = input.parse()?;
39            return Ok(PreAuthorizeArgs {
40                check: PreAuthorizeCheck::Expression(lit.value()),
41            });
42        }
43
44        // Check for bare "authenticated" identifier
45        if input.peek(syn::Ident) {
46            let ident: syn::Ident = input.parse()?;
47
48            if ident == "authenticated" {
49                return Ok(PreAuthorizeArgs {
50                    check: PreAuthorizeCheck::Authenticated,
51                });
52            }
53
54            // Check for key = value syntax
55            if input.peek(Token![=]) {
56                input.parse::<Token![=]>()?;
57
58                match ident.to_string().as_str() {
59                    "role" | "roles" => {
60                        let roles = parse_string_or_array(input)?;
61                        return Ok(PreAuthorizeArgs {
62                            check: PreAuthorizeCheck::Roles(roles),
63                        });
64                    }
65                    "authority" | "authorities" => {
66                        let authorities = parse_string_or_array(input)?;
67                        return Ok(PreAuthorizeArgs {
68                            check: PreAuthorizeCheck::Authorities(authorities),
69                        });
70                    }
71                    _ => {
72                        return Err(syn::Error::new_spanned(
73                            ident,
74                            "expected 'authenticated', 'role', 'roles', 'authority', or 'authorities'",
75                        ));
76                    }
77                }
78            }
79
80            return Err(syn::Error::new_spanned(
81                ident,
82                "expected 'authenticated' or 'key = value' syntax",
83            ));
84        }
85
86        Err(syn::Error::new(
87            input.span(),
88            r#"expected one of: "expression", authenticated, role = "...", authority = "...""#,
89        ))
90    }
91}

93/// Parses either a single string "value" or an array ["a", "b"]
94fn parse_string_or_array(input: ParseStream) -> syn::Result<Vec<String>> {
95    if input.peek(syn::token::Bracket) {
96        // Array syntax: ["a", "b"]
97        let content;
98        syn::bracketed!(content in input);
99        let args: Punctuated<Expr, Token![,]> = Punctuated::parse_terminated(&content)?;
100
101        let mut values = Vec::new();
102        for arg in args {
103            if let Expr::Lit(ExprLit {
104                lit: Lit::Str(lit_str),
105                ..
106            }) = arg
107            {
108                values.push(lit_str.value());
109            } else {
110                return Err(syn::Error::new_spanned(arg, "expected string literal"));
111            }
112        }
113        Ok(values)
114    } else {
115        // Single string: "value"
116        let lit: syn::LitStr = input.parse()?;
117        Ok(vec![lit.value()])
118    }
119}

125/// Information about a handler parameter that can be referenced in expressions
126#[derive(Debug, Clone)]
127#[allow(dead_code)] // inner_type and tuple_index reserved for future tuple support
128struct HandlerParam {
129    /// The parameter name (e.g., "tenant_id")
130    name: String,
131    /// The extractor type (e.g., "Path", "Query", "Json")
132    extractor_type: String,
133    /// The inner type as a string (e.g., "i64", "(i64, String)")
134    inner_type: String,
135    /// For tuple extractors, the position of this param (0-indexed)
136    tuple_index: Option<usize>,
137    /// The full parameter identifier
138    ident: proc_macro2::Ident,
139}
140
141/// Extracts information about handler parameters that can be referenced
142fn extract_handler_params(item_fn: &ItemFn) -> Vec<HandlerParam> {
143    let mut params = Vec::new();
144
145    for arg in &item_fn.sig.inputs {
146        if let FnArg::Typed(pat_type) = arg {
147            // Get the parameter name
148            let param_name = if let Pat::Ident(pat_ident) = pat_type.pat.as_ref() {
149                pat_ident.ident.to_string()
150            } else {
151                continue;
152            };
153
154            let param_ident = if let Pat::Ident(pat_ident) = pat_type.pat.as_ref() {
155                pat_ident.ident.clone()
156            } else {
157                continue;
158            };
159
160            // Check if this is an extractor type (Path, Query, Json, etc.)
161            if let Type::Path(type_path) = pat_type.ty.as_ref() {
162                if let Some(segment) = type_path.path.segments.last() {
163                    let type_name = segment.ident.to_string();
164
165                    // Check for supported extractor types
166                    if matches!(type_name.as_str(), "Path" | "Query" | "Json" | "Form") {
167                        // Extract the inner type
168                        if let PathArguments::AngleBracketed(args) = &segment.arguments {
169                            if let Some(GenericArgument::Type(inner_ty)) = args.args.first() {
170                                let inner_type_str = quote::quote!(#inner_ty).to_string();
171
172                                // Check if it's a tuple type
173                                if let Type::Tuple(tuple) = inner_ty {
174                                    // For tuples, we can't automatically map param names
175                                    // The user would need to use index syntax or we match by name
176                                    // For now, store the whole tuple info
177                                    params.push(HandlerParam {
178                                        name: param_name,
179                                        extractor_type: type_name,
180                                        inner_type: inner_type_str,
181                                        tuple_index: None,
182                                        ident: param_ident,
183                                    });
184
185                                    // Note: For tuple support, we'd need additional syntax
186                                    // like #path.0 or explicit naming in the Path<...>
187                                    let _ = tuple; // Acknowledge we see the tuple
188                                } else {
189                                    // Single value extractor
190                                    params.push(HandlerParam {
191                                        name: param_name,
192                                        extractor_type: type_name,
193                                        inner_type: inner_type_str,
194                                        tuple_index: None,
195                                        ident: param_ident,
196                                    });
197                                }
198                            }
199                        }
200                    }
201                }
202            }
203        }
204    }
205
206    params
207}

209/// Finds a handler parameter by name and generates code to extract its value
210fn generate_param_extraction(
211    param_name: &str,
212    handler_params: &[HandlerParam],
213) -> Result<TokenStream2, String> {
214    // First, look for a direct match by parameter name
215    for param in handler_params {
216        if param.name == param_name {
217            let param_ident = &param.ident;
218            let param_var = format_ident!("__param_{}", param_name);
219
220            // Generate extraction code based on extractor type
221            // Note: We dereference the extractor to get the inner value
222            return match param.extractor_type.as_str() {
223                "Path" | "Query" | "Json" | "Form" => {
224                    // Use deref (*) to get the inner value, then clone it
225                    Ok(quote! {
226                        let #param_var = (*#param_ident).clone();
227                    })
228                }
229                _ => Err(format!(
230                    "unsupported extractor type '{}' for parameter '{}'",
231                    param.extractor_type, param_name
232                )),
233            };
234        }
235    }
236
237    Err(format!(
238        "parameter '{}' not found in handler. Available parameters: {}",
239        param_name,
240        handler_params
241            .iter()
242            .map(|p| p.name.as_str())
243            .collect::<Vec<_>>()
244            .join(", ")
245    ))
246}

252/// Compiles a security expression to Rust code.
253/// Returns the parsed expression with code and metadata.
254fn compile_expression(
255    expr: &str,
256    user_param: &proc_macro2::Ident,
257) -> Result<ParsedExpression, String> {
258    let tokens = tokenize_expression(expr)?;
259    if tokens.is_empty() {
260        return Err("empty expression".to_string());
261    }
262
263    let parser = ExpressionParser::new(tokens);
264    parser.parse(user_param)
265}

267#[derive(Debug, Clone, PartialEq)]
268enum ExprToken {
269    Ident(String),
270    String(String),
271    /// Parameter reference: #param_name
272    ParamRef(String),
273    LParen,
274    RParen,
275    Comma,
276    And,
277    Or,
278    Not,
279    True,
280    False,
281}

283/// Represents a function argument - either a string literal or a parameter reference
284#[derive(Debug, Clone)]
285enum FunctionArg {
286    /// A string literal argument: 'value'
287    StringLit(String),
288    /// A parameter reference: #param_name
289    ParamRef(String),
290}
291
292fn tokenize_expression(expr: &str) -> Result<Vec<ExprToken>, String> {
293    let mut tokens = Vec::new();
294    let mut chars = expr.chars().peekable();
295
296    while let Some(&c) = chars.peek() {
297        match c {
298            ' ' | '\t' | '\n' | '\r' => {
299                chars.next();
300            }
301            '(' => {
302                chars.next();
303                tokens.push(ExprToken::LParen);
304            }
305            ')' => {
306                chars.next();
307                tokens.push(ExprToken::RParen);
308            }
309            ',' => {
310                chars.next();
311                tokens.push(ExprToken::Comma);
312            }
313            '\'' | '"' => {
314                let quote = chars.next().unwrap();
315                let mut value = String::new();
316                loop {
317                    match chars.next() {
318                        Some(c) if c == quote => break,
319                        Some('\\') => {
320                            if let Some(escaped) = chars.next() {
321                                value.push(escaped);
322                            }
323                        }
324                        Some(c) => value.push(c),
325                        None => return Err("unclosed string literal".to_string()),
326                    }
327                }
328                tokens.push(ExprToken::String(value));
329            }
330            '&' => {
331                chars.next();
332                if chars.peek() == Some(&'&') {
333                    chars.next();
334                    tokens.push(ExprToken::And);
335                } else {
336                    return Err("expected && operator".to_string());
337                }
338            }
339            '|' => {
340                chars.next();
341                if chars.peek() == Some(&'|') {
342                    chars.next();
343                    tokens.push(ExprToken::Or);
344                } else {
345                    return Err("expected || operator".to_string());
346                }
347            }
348            '!' => {
349                chars.next();
350                tokens.push(ExprToken::Not);
351            }
352            // Parameter reference: #param_name
353            '#' => {
354                chars.next();
355                let mut param_name = String::new();
356                while let Some(&c) = chars.peek() {
357                    if c.is_alphanumeric() || c == '_' {
358                        param_name.push(c);
359                        chars.next();
360                    } else {
361                        break;
362                    }
363                }
364                if param_name.is_empty() {
365                    return Err("expected parameter name after '#'".to_string());
366                }
367                tokens.push(ExprToken::ParamRef(param_name));
368            }
369            'a'..='z' | 'A'..='Z' | '_' => {
370                let mut ident = String::new();
371                while let Some(&c) = chars.peek() {
372                    if c.is_alphanumeric() || c == '_' {
373                        ident.push(c);
374                        chars.next();
375                    } else {
376                        break;
377                    }
378                }
379                match ident.to_lowercase().as_str() {
380                    "and" => tokens.push(ExprToken::And),
381                    "or" => tokens.push(ExprToken::Or),
382                    "not" => tokens.push(ExprToken::Not),
383                    "true" => tokens.push(ExprToken::True),
384                    "false" => tokens.push(ExprToken::False),
385                    _ => tokens.push(ExprToken::Ident(ident)),
386                }
387            }
388            _ => return Err(format!("unexpected character: '{}'", c)),
389        }
390    }
391
392    Ok(tokens)
393}

395/// Result of parsing an expression - contains the code and any parameter references
396struct ParsedExpression {
397    /// The generated Rust code for the expression
398    code: TokenStream2,
399    /// Parameter references found in the expression (for custom functions)
400    param_refs: Vec<String>,
401    /// Whether the expression contains custom async functions
402    has_custom_functions: bool,
403}
404
405struct ExpressionParser {
406    tokens: Vec<ExprToken>,
407    pos: usize,
408    /// Collected parameter references from custom function calls
409    param_refs: Vec<String>,
410    /// Whether we've seen any custom (non-builtin) functions
411    has_custom_functions: bool,
412}
413
414impl ExpressionParser {
415    fn new(tokens: Vec<ExprToken>) -> Self {
416        ExpressionParser {
417            tokens,
418            pos: 0,
419            param_refs: Vec::new(),
420            has_custom_functions: false,
421        }
422    }
423
424    fn peek(&self) -> Option<&ExprToken> {
425        self.tokens.get(self.pos)
426    }
427
428    fn advance(&mut self) -> Option<ExprToken> {
429        let token = self.tokens.get(self.pos).cloned();
430        self.pos += 1;
431        token
432    }
433
434    fn parse(mut self, user: &proc_macro2::Ident) -> Result<ParsedExpression, String> {
435        let code = self.parse_or(user)?;
436        Ok(ParsedExpression {
437            code,
438            param_refs: self.param_refs,
439            has_custom_functions: self.has_custom_functions,
440        })
441    }
442
443    fn parse_or(&mut self, user: &proc_macro2::Ident) -> Result<TokenStream2, String> {
444        let mut left = self.parse_and(user)?;
445
446        while matches!(self.peek(), Some(ExprToken::Or)) {
447            self.advance();
448            let right = self.parse_and(user)?;
449            left = quote! { (#left) || (#right) };
450        }
451
452        Ok(left)
453    }
454
455    fn parse_and(&mut self, user: &proc_macro2::Ident) -> Result<TokenStream2, String> {
456        let mut left = self.parse_unary(user)?;
457
458        while matches!(self.peek(), Some(ExprToken::And)) {
459            self.advance();
460            let right = self.parse_unary(user)?;
461            left = quote! { (#left) && (#right) };
462        }
463
464        Ok(left)
465    }
466
467    fn parse_unary(&mut self, user: &proc_macro2::Ident) -> Result<TokenStream2, String> {
468        if matches!(self.peek(), Some(ExprToken::Not)) {
469            self.advance();
470            let expr = self.parse_unary(user)?;
471            return Ok(quote! { !(#expr) });
472        }
473
474        self.parse_primary(user)
475    }
476
477    fn parse_primary(&mut self, user: &proc_macro2::Ident) -> Result<TokenStream2, String> {
478        match self.peek().cloned() {
479            Some(ExprToken::True) => {
480                self.advance();
481                Ok(quote! { true })
482            }
483            Some(ExprToken::False) => {
484                self.advance();
485                Ok(quote! { false })
486            }
487            Some(ExprToken::LParen) => {
488                self.advance();
489                let expr = self.parse_or(user)?;
490                if !matches!(self.peek(), Some(ExprToken::RParen)) {
491                    return Err("unclosed parenthesis".to_string());
492                }
493                self.advance();
494                Ok(quote! { (#expr) })
495            }
496            Some(ExprToken::Ident(name)) => {
497                self.advance();
498                self.parse_function_call(&name, user)
499            }
500            Some(token) => Err(format!("unexpected token: {:?}", token)),
501            None => Err("unexpected end of expression".to_string()),
502        }
503    }
504
505    fn parse_function_call(
506        &mut self,
507        name: &str,
508        user: &proc_macro2::Ident,
509    ) -> Result<TokenStream2, String> {
510        // Expect opening parenthesis
511        if !matches!(self.peek(), Some(ExprToken::LParen)) {
512            return Err(format!("expected '(' after function name '{}'", name));
513        }
514        self.advance();
515
516        // Parse arguments (can be strings, identifiers, or param refs)
517        let mut args: Vec<FunctionArg> = Vec::new();
518        if !matches!(self.peek(), Some(ExprToken::RParen)) {
519            loop {
520                match self.peek().cloned() {
521                    Some(ExprToken::String(s)) => {
522                        self.advance();
523                        args.push(FunctionArg::StringLit(s));
524                    }
525                    Some(ExprToken::Ident(s)) => {
526                        self.advance();
527                        args.push(FunctionArg::StringLit(s));
528                    }
529                    Some(ExprToken::ParamRef(param_name)) => {
530                        self.advance();
531                        // Track this parameter reference
532                        if !self.param_refs.contains(&param_name) {
533                            self.param_refs.push(param_name.clone());
534                        }
535                        args.push(FunctionArg::ParamRef(param_name));
536                    }
537                    Some(token) => {
538                        return Err(format!(
539                            "unexpected token in function arguments: {:?}",
540                            token
541                        ))
542                    }
543                    None => return Err("unclosed function call".to_string()),
544                }
545
546                match self.peek() {
547                    Some(ExprToken::Comma) => {
548                        self.advance();
549                    }
550                    Some(ExprToken::RParen) => break,
551                    Some(token) => {
552                        return Err(format!("unexpected token: {:?}", token));
553                    }
554                    None => return Err("unclosed function call".to_string()),
555                }
556            }
557        }
558
559        // Expect closing parenthesis
560        if !matches!(self.peek(), Some(ExprToken::RParen)) {
561            return Err("unclosed function call".to_string());
562        }
563        self.advance();
564
565        // Check if this is a built-in function or custom function
566        if Self::is_builtin_function(name) {
567            self.generate_builtin_function_code(name, &args, user)
568        } else {
569            self.has_custom_functions = true;
570            self.generate_custom_function_code(name, &args, user)
571        }
572    }
573
574    fn is_builtin_function(name: &str) -> bool {
575        matches!(
576            name,
577            // camelCase (Spring Security style)
578            "hasRole"
579                | "hasAnyRole"
580                | "hasAuthority"
581                | "hasAnyAuthority"
582                | "isAuthenticated"
583                | "permitAll"
584                | "denyAll"
585                // snake_case (Rust style)
586                | "has_role"
587                | "has_any_role"
588                | "has_authority"
589                | "has_any_authority"
590                | "is_authenticated"
591                | "permit_all"
592                | "deny_all"
593        )
594    }
595
596    /// Normalizes a built-in function name to its canonical form for code generation.
597    /// Both camelCase and snake_case are accepted.
598    fn normalize_builtin_function(name: &str) -> &'static str {
599        match name {
600            "hasRole" | "has_role" => "hasRole",
601            "hasAnyRole" | "has_any_role" => "hasAnyRole",
602            "hasAuthority" | "has_authority" => "hasAuthority",
603            "hasAnyAuthority" | "has_any_authority" => "hasAnyAuthority",
604            "isAuthenticated" | "is_authenticated" => "isAuthenticated",
605            "permitAll" | "permit_all" => "permitAll",
606            "denyAll" | "deny_all" => "denyAll",
607            // This should never be reached if is_builtin_function was called first
608            _ => unreachable!(
609                "normalize_builtin_function called with non-builtin: {}",
610                name
611            ),
612        }
613    }
614
615    fn generate_builtin_function_code(
616        &self,
617        name: &str,
618        args: &[FunctionArg],
619        user: &proc_macro2::Ident,
620    ) -> Result<TokenStream2, String> {
621        // Extract string literals for built-in functions (they don't support param refs)
622        let string_args: Result<Vec<&str>, String> = args
623            .iter()
624            .map(|arg| match arg {
625                FunctionArg::StringLit(s) => Ok(s.as_str()),
626                FunctionArg::ParamRef(p) => Err(format!(
627                    "built-in function '{}' does not support parameter references (got #{})",
628                    name, p
629                )),
630            })
631            .collect();
632        let string_args = string_args?;
633
634        // Normalize function name to handle both camelCase and snake_case
635        let canonical_name = Self::normalize_builtin_function(name);
636
637        match canonical_name {
638            "hasRole" => {
639                let role = string_args
640                    .first()
641                    .ok_or_else(|| "hasRole requires a role argument".to_string())?;
642                Ok(quote! { #user.has_role(#role) })
643            }
644            "hasAnyRole" => {
645                if string_args.is_empty() {
646                    return Err("hasAnyRole requires at least one role argument".to_string());
647                }
648                Ok(quote! { #user.has_any_role(&[#(#string_args),*]) })
649            }
650            "hasAuthority" => {
651                let authority = string_args
652                    .first()
653                    .ok_or_else(|| "hasAuthority requires an authority argument".to_string())?;
654                Ok(quote! { #user.has_authority(#authority) })
655            }
656            "hasAnyAuthority" => {
657                if string_args.is_empty() {
658                    return Err(
659                        "hasAnyAuthority requires at least one authority argument".to_string()
660                    );
661                }
662                Ok(quote! { #user.has_any_authority(&[#(#string_args),*]) })
663            }
664            "isAuthenticated" => {
665                // AuthenticatedUser extractor guarantees authentication
666                Ok(quote! { true })
667            }
668            "permitAll" => Ok(quote! { true }),
669            "denyAll" => Ok(quote! { false }),
670            _ => unreachable!("is_builtin_function should have caught this"),
671        }
672    }
673
674    fn generate_custom_function_code(
675        &self,
676        name: &str,
677        args: &[FunctionArg],
678        user: &proc_macro2::Ident,
679    ) -> Result<TokenStream2, String> {
680        let func_ident = format_ident!("{}", name);
681
682        // Generate argument tokens
683        let arg_tokens: Vec<TokenStream2> = args
684            .iter()
685            .map(|arg| match arg {
686                FunctionArg::StringLit(s) => quote! { #s },
687                FunctionArg::ParamRef(param_name) => {
688                    let param_var = format_ident!("__param_{}", param_name);
689                    quote! { #param_var }
690                }
691            })
692            .collect();
693
694        // Custom functions are called with user as first argument, then the other args
695        // The function signature should be: async fn name(user: &User, ...) -> bool
696        // Use &* to dereference AuthenticatedUser (which implements Deref<Target=User>)
697        Ok(quote! { #func_ident(&*#user, #(#arg_tokens),*).await })
698    }
699}

701/// Flexible method security annotation with SpEL-like expressions.
702///
703/// # Spring Security Equivalent
704/// `@PreAuthorize("...")`
705///
706/// # Usage
707/// ```ignore
708/// use actix_security::http::security::AuthenticatedUser;
709/// use actix_security_codegen::pre_authorize;
710///
711/// // Expression syntax (Spring Security-like)
712/// #[pre_authorize("hasRole('ADMIN') OR hasAuthority('users:write')")]
713/// #[get("/admin")]
714/// async fn admin(user: AuthenticatedUser) -> impl Responder {
715///     HttpResponse::Ok().body("Admin")
716/// }
717///
718/// // Complex expressions with AND, OR, NOT
719/// #[pre_authorize("(hasRole('ADMIN') OR hasRole('MANAGER')) AND hasAuthority('users:read')")]
720/// #[get("/management")]
721/// async fn management(user: AuthenticatedUser) -> impl Responder {
722///     HttpResponse::Ok().body("Management")
723/// }
724///
725/// // Check authentication only
726/// #[pre_authorize(authenticated)]
727/// #[get("/protected")]
728/// async fn protected(user: AuthenticatedUser) -> impl Responder {
729///     HttpResponse::Ok().body("Protected")
730/// }
731///
732/// // Check single role (shorthand)
733/// #[pre_authorize(role = "ADMIN")]
734/// #[get("/admin-only")]
735/// async fn admin_only(user: AuthenticatedUser) -> impl Responder {
736///     HttpResponse::Ok().body("Admin Only")
737/// }
738///
739/// // Check multiple roles (OR logic)
740/// #[pre_authorize(roles = ["ADMIN", "MANAGER"])]
741/// #[get("/managers")]
742/// async fn managers(user: AuthenticatedUser) -> impl Responder {
743///     HttpResponse::Ok().body("Managers")
744/// }
745///
746/// // Check authority
747/// #[pre_authorize(authority = "users:read")]
748/// #[get("/api/users")]
749/// async fn get_users(user: AuthenticatedUser) -> impl Responder {
750///     HttpResponse::Ok().body("Users")
751/// }
752///
753/// // Check multiple authorities (OR logic)
754/// #[pre_authorize(authorities = ["users:read", "users:write"])]
755/// #[get("/api/users/manage")]
756/// async fn manage_users(user: AuthenticatedUser) -> impl Responder {
757///     HttpResponse::Ok().body("Manage users")
758/// }
759/// ```
760///
761/// # Custom Authorization Functions
762///
763/// You can use custom async functions with parameter references (Spring Security style):
764///
765/// ```ignore
766/// // Define your custom authorization function
767/// // It must be async and return bool, with user as first parameter
768/// pub async fn is_tenant_admin(user: &User, tenant_id: i64) -> bool {
769///     user.has_authority(&format!("tenant:{}:admin", tenant_id))
770/// }
771///
772/// // Use #param_name to reference handler parameters
773/// #[pre_authorize("is_tenant_admin(#tenant_id)")]
774/// #[get("/tenants/{tenant_id}")]
775/// async fn get_tenant(
776///     tenant_id: Path<i64>,
777///     user: AuthenticatedUser,
778/// ) -> impl Responder {
779///     HttpResponse::Ok().body("Tenant data")
780/// }
781///
782/// // Combine with built-in functions
783/// #[pre_authorize("hasRole('ADMIN') OR is_tenant_admin(#tenant_id)")]
784/// #[get("/tenants/{tenant_id}/settings")]
785/// async fn get_tenant_settings(
786///     tenant_id: Path<i64>,
787///     user: AuthenticatedUser,
788/// ) -> impl Responder {
789///     HttpResponse::Ok().body("Tenant settings")
790/// }
791///
792/// // Multiple parameters
793/// pub async fn can_access_document(user: &User, tenant_id: i64, doc_id: String) -> bool {
794///     // Custom authorization logic
795///     true
796/// }
797///
798/// #[pre_authorize("can_access_document(#tenant_id, #doc_id)")]
799/// #[get("/tenants/{tenant_id}/documents/{doc_id}")]
800/// async fn get_document(
801///     tenant_id: Path<i64>,
802///     doc_id: Path<String>,
803///     user: AuthenticatedUser,
804/// ) -> impl Responder {
805///     HttpResponse::Ok().body("Document")
806/// }
807/// ```
808///
809/// ## Supported Extractor Types for #param
810///
811/// - `Path<T>` - URL path parameters
812/// - `Query<T>` - Query string parameters
813/// - `Json<T>` - JSON request body
814/// - `Form<T>` - Form data
815///
816/// # Supported Expression Functions
817///
818/// ## Built-in Functions
819/// - `hasRole('ROLE')` - Check if user has the specified role
820/// - `hasAnyRole('ROLE1', 'ROLE2')` - Check if user has any of the roles
821/// - `hasAuthority('AUTH')` - Check if user has the specified authority
822/// - `hasAnyAuthority('AUTH1', 'AUTH2')` - Check if user has any authority
823/// - `isAuthenticated()` - Check if user is authenticated (always true with AuthenticatedUser)
824/// - `permitAll()` - Always returns true
825/// - `denyAll()` - Always returns false
826///
827/// ## Custom Functions
828/// Any function matching signature: `async fn name(user: &User, ...) -> bool`
829///
830/// # Operators
831///
832/// - `AND` / `and` / `&&` - Logical AND
833/// - `OR` / `or` / `||` - Logical OR
834/// - `NOT` / `not` / `!` - Logical NOT
835/// - `(` `)` - Grouping for precedence
836pub fn pre_authorize_impl(attrs: TokenStream, input: TokenStream) -> TokenStream {
837    let item_fn = parse_macro_input!(input as ItemFn);
838    let args = parse_macro_input!(attrs as PreAuthorizeArgs);
839
840    let user_param = match find_user_param(&item_fn) {
841        Some(param) => param,
842        None => return missing_user_param_error(&item_fn, "pre_authorize"),
843    };
844
845    // Extract handler parameters for potential #param references
846    let handler_params = extract_handler_params(&item_fn);
847
848    let attrs = &item_fn.attrs;
849    let vis = &item_fn.vis;
850    let sig = &item_fn.sig;
851    let block = &item_fn.block;
852    let fn_name = &sig.ident;
853    let inputs = &sig.inputs;
854    let asyncness = &sig.asyncness;
855    let generics = &sig.generics;
856
857    let original_return = match &sig.output {
858        ReturnType::Default => quote! { () },
859        ReturnType::Type(_, ty) => quote! { #ty },
860    };
861
862    let core_path = core_crate_path();
863
864    let check_code = match args.check {
865        PreAuthorizeCheck::Authenticated => {
866            // AuthenticatedUser extractor already handles this, but we keep it for explicitness
867            quote! {
868                // Authentication check (handled by AuthenticatedUser extractor)
869                let _ = &#user_param;
870            }
871        }
872        PreAuthorizeCheck::Roles(roles) => {
873            quote! {
874                // Role check generated by #[pre_authorize(role = ...)]
875                let __required_roles: &[&str] = &[#(#roles),*];
876                if !#user_param.has_any_role(__required_roles) {
877                    return ::std::result::Result::Err(#core_path::http::error::AuthError::Forbidden);
878                }
879            }
880        }
881        PreAuthorizeCheck::Authorities(authorities) => {
882            quote! {
883                // Authority check generated by #[pre_authorize(authority = ...)]
884                let __required_authorities: &[&str] = &[#(#authorities),*];
885                if !#user_param.has_any_authority(__required_authorities) {
886                    return ::std::result::Result::Err(#core_path::http::error::AuthError::Forbidden);
887                }
888            }
889        }
890        PreAuthorizeCheck::Expression(expr_str) => {
891            match compile_expression(&expr_str, &user_param) {
892                Ok(parsed_expr) => {
893                    let expr_code = parsed_expr.code;
894
895                    // If expression has custom functions with param refs, generate extraction code
896                    if parsed_expr.has_custom_functions && !parsed_expr.param_refs.is_empty() {
897                        // Generate extraction code for each referenced parameter
898                        let mut extraction_code = Vec::new();
899                        for param_ref in &parsed_expr.param_refs {
900                            match generate_param_extraction(param_ref, &handler_params) {
901                                Ok(code) => extraction_code.push(code),
902                                Err(err) => {
903                                    let err_msg = format!("invalid parameter reference: {}", err);
904                                    return syn::Error::new_spanned(&item_fn.sig.ident, err_msg)
905                                        .to_compile_error()
906                                        .into();
907                                }
908                            }
909                        }
910
911                        quote! {
912                            // Parameter extraction for custom authorization function
913                            #(#extraction_code)*
914                            // Expression check generated by #[pre_authorize("...")]
915                            if !(#expr_code) {
916                                return ::std::result::Result::Err(#core_path::http::error::AuthError::Forbidden);
917                            }
918                        }
919                    } else {
920                        quote! {
921                            // Expression check generated by #[pre_authorize("...")]
922                            if !(#expr_code) {
923                                return ::std::result::Result::Err(#core_path::http::error::AuthError::Forbidden);
924                            }
925                        }
926                    }
927                }
928                Err(err) => {
929                    let err_msg = format!("invalid security expression: {}", err);
930                    return syn::Error::new_spanned(&item_fn.sig.ident, err_msg)
931                        .to_compile_error()
932                        .into();
933                }
934            }
935        }
936    };
937
938    let expanded = quote! {
939        #(#attrs)*
940        #vis #asyncness fn #fn_name #generics(#inputs) -> ::std::result::Result<#original_return, #core_path::http::error::AuthError> {
941            {
942                #check_code
943            }
944
945            ::std::result::Result::Ok(#block)
946        }
947    };
948
949    expanded.into()
950}
