1//! The `#[secured]` macro for role-based method security.
2//!
3//! # Spring Security Equivalent
4//! `@Secured("ROLE_ADMIN")` or `@RolesAllowed("ADMIN")`
5
6use proc_macro::TokenStream;
7use quote::quote;
8use syn::parse::{Parse, ParseStream};
9use syn::punctuated::Punctuated;
10use syn::{parse_macro_input, Expr, ExprLit, ItemFn, Lit, ReturnType, Token};
11
12use crate::helpers::{find_user_param, missing_user_param_error};
13
14/// Parses comma-separated string literals for #[secured("ADMIN", "USER")]
15pub struct SecuredArgs {
16    pub roles: Vec<String>,
17}
18
19impl Parse for SecuredArgs {
20    fn parse(input: ParseStream) -> syn::Result<Self> {
21        let args: Punctuated<Expr, Token![,]> = Punctuated::parse_terminated(input)?;
22
23        if args.is_empty() {
24            return Err(syn::Error::new(
25                input.span(),
26                r#"expected at least one role, e.g. #[secured("ADMIN")]"#,
27            ));
28        }
29
30        let mut roles = Vec::new();
31        for arg in args {
32            match arg {
33                Expr::Lit(ExprLit {
34                    lit: Lit::Str(lit_str),
35                    ..
36                }) => {
37                    roles.push(lit_str.value());
38                }
39                _ => {
40                    return Err(syn::Error::new_spanned(
41                        arg,
42                        r#"expected string literal like "ADMIN""#,
43                    ));
44                }
45            }
46        }
47
48        Ok(SecuredArgs { roles })
49    }
50}
51
52/// Role-based method security annotation.
53///
54/// # Spring Security Equivalent
55/// `@Secured("ROLE_ADMIN")` or `@RolesAllowed("ADMIN")`
56///
57/// # Usage
58/// ```ignore
59/// use actix_security_core::http::security::AuthenticatedUser;
60/// use actix_security_codegen::secured;
61///
62/// // Single role
63/// #[secured("ADMIN")]
64/// #[get("/admin")]
65/// async fn admin_only(user: AuthenticatedUser) -> impl Responder {
66///     HttpResponse::Ok().body("Admin area")
67/// }
68///
69/// // Multiple roles (OR logic - user needs ANY of these)
70/// #[secured("ADMIN", "MANAGER")]
71/// #[get("/management")]
72/// async fn management(user: AuthenticatedUser) -> impl Responder {
73///     HttpResponse::Ok().body("Management area")
74/// }
75/// ```
76///
77/// # Note
78/// Unlike Spring Security, you don't need the "ROLE_" prefix.
79/// The macro checks if the user has ANY of the specified roles.
80pub fn secured_impl(attrs: TokenStream, input: TokenStream) -> TokenStream {
81    let item_fn = parse_macro_input!(input as ItemFn);
82    let args = parse_macro_input!(attrs as SecuredArgs);
83
84    let roles = args.roles;
85
86    let user_param = match find_user_param(&item_fn) {
87        Some(param) => param,
88        None => return missing_user_param_error(&item_fn, "secured"),
89    };
90
91    let attrs = &item_fn.attrs;
92    let vis = &item_fn.vis;
93    let sig = &item_fn.sig;
94    let block = &item_fn.block;
95    let fn_name = &sig.ident;
96    let inputs = &sig.inputs;
97    let asyncness = &sig.asyncness;
98    let generics = &sig.generics;
99
100    let original_return = match &sig.output {
101        ReturnType::Default => quote! { () },
102        ReturnType::Type(_, ty) => quote! { #ty },
103    };
104
105    let expanded = quote! {
106        #(#attrs)*
107        #vis #asyncness fn #fn_name #generics(#inputs) -> ::std::result::Result<#original_return, ::actix_security_core::http::error::AuthError> {
108            // Role check generated by #[secured]
109            {
110                let __required_roles: &[&str] = &[#(#roles),*];
111                if !#user_param.has_any_role(__required_roles) {
112                    return ::std::result::Result::Err(::actix_security_core::http::error::AuthError::Forbidden);
113                }
114            }
115
116            ::std::result::Result::Ok(#block)
117        }
118    };
119
120    expanded.into()
121}
