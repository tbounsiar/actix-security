1//! LDAP Authentication Example
2//!
3//! This example demonstrates how to configure LDAP authentication.
4//! It uses a mock LDAP client for demonstration purposes.
5//!
6//! ## Running the example
7//!
8//! ```bash
9//! cargo run -p actix-security-examples --bin ldap_auth
10//! ```
11//!
12//! ## Testing with curl
13//!
14//! ```bash
15//! # Test users (mock LDAP)
16//! curl -u john.doe:password http://127.0.0.1:8080/
17//! curl -u jane.smith:secret http://127.0.0.1:8080/
18//! curl -u admin:admin123 http://127.0.0.1:8080/admin
19//!
20//! # Invalid credentials
21//! curl -u invalid:wrong http://127.0.0.1:8080/
22//! ```

80fn get_mock_client() -> Arc<MockLdapClient> {
81    MOCK_CLIENT
82        .get_or_init(|| {
83            let mock = MockLdapClient::new();
84
85            // Add user: john.doe (regular user)
86            let mut john_attrs = HashMap::new();
87            john_attrs.insert("cn".to_string(), vec!["John Doe".to_string()]);
88            john_attrs.insert("mail".to_string(), vec!["john.doe@example.com".to_string()]);
89            mock.add_user(
90                "john.doe",
91                "password",
92                john_attrs,
93                vec!["cn=users,ou=groups,dc=example,dc=com".to_string()],
94            );
95
96            // Add user: jane.smith (manager)
97            let mut jane_attrs = HashMap::new();
98            jane_attrs.insert("cn".to_string(), vec!["Jane Smith".to_string()]);
99            jane_attrs.insert(
100                "mail".to_string(),
101                vec!["jane.smith@example.com".to_string()],
102            );
103            mock.add_user(
104                "jane.smith",
105                "secret",
106                jane_attrs,
107                vec![
108                    "cn=users,ou=groups,dc=example,dc=com".to_string(),
109                    "cn=managers,ou=groups,dc=example,dc=com".to_string(),
110                ],
111            );
112
113            // Add user: admin (administrator)
114            let mut admin_attrs = HashMap::new();
115            admin_attrs.insert("cn".to_string(), vec!["Administrator".to_string()]);
116            admin_attrs.insert("mail".to_string(), vec!["admin@example.com".to_string()]);
117            mock.add_user(
118                "admin",
119                "admin123",
120                admin_attrs,
121                vec![
122                    "cn=users,ou=groups,dc=example,dc=com".to_string(),
123                    "cn=admins,ou=groups,dc=example,dc=com".to_string(),
124                ],
125            );
126
127            Arc::new(mock)
128        })
129        .clone()
130}

132fn get_ldap_config() -> Arc<LdapConfig> {
133    LDAP_CONFIG
134        .get_or_init(|| {
135            Arc::new(
136                LdapConfig::new("ldap://localhost:389")
137                    .base_dn("dc=example,dc=com")
138                    .user_search_base("ou=users")
139                    .user_search_filter("(uid={0})")
140                    .group_search_base("ou=groups")
141                    .group_search_filter("(member={0})")
142                    .role_prefix("ROLE_")
143                    .convert_to_uppercase(true),
144            )
145        })
146        .clone()
147}

153/// Simple LDAP-based authenticator using mock client
154#[derive(Clone)]
155struct MockLdapAuthenticator;
156
157impl Authenticator for MockLdapAuthenticator {
158    fn get_user(&self, req: &ServiceRequest) -> Option<User> {
159        // Extract HTTP Basic credentials
160        let auth_header = req.headers().get("Authorization")?;
161        let auth_str = auth_header.to_str().ok()?;
162
163        if !auth_str.starts_with("Basic ") {
164            return None;
165        }
166
167        let credentials = &auth_str[6..];
168        let decoded = base64_decode(credentials)?;
169        let decoded_str = String::from_utf8(decoded).ok()?;
170        let parts: Vec<&str> = decoded_str.splitn(2, ':').collect();
171
172        if parts.len() != 2 {
173            return None;
174        }
175
176        let username = parts[0];
177        let password = parts[1];
178
179        // Get shared state
180        let mock = get_mock_client();
181        let config = get_ldap_config();
182
183        // For this example, we use a simple blocking approach
184        // In production, you'd want to use an async-aware authenticator
185        use actix_security::http::security::ldap::LdapOperations;
186
187        let username_str = username.to_string();
188        let password_str = password.to_string();
189
190        let result = std::thread::scope(|s| {
191            s.spawn(|| {
192                let rt = actix_rt::Runtime::new().unwrap();
193                rt.block_on(async { mock.authenticate(&username_str, &password_str).await })
194            })
195            .join()
196            .unwrap()
197        });
198
199        match result {
200            Ok(auth_result) if auth_result.success => {
201                // Build user from LDAP result
202                let roles: Vec<String> = auth_result
203                    .groups
204                    .iter()
205                    .filter_map(|group_dn: &String| {
206                        group_dn
207                            .split(',')
208                            .next()
209                            .and_then(|cn_part: &str| cn_part.strip_prefix("cn="))
210                            .map(|cn: &str| {
211                                let role = if config.convert_to_uppercase {
212                                    cn.to_uppercase()
213                                } else {
214                                    cn.to_string()
215                                };
216                                format!("{}{}", config.role_prefix, role)
217                            })
218                    })
219                    .collect();
220
221                Some(User::new(username_str, String::new()).roles(&roles))
222            }
223            _ => None,
224        }
225    }
226}

228fn base64_decode(input: &str) -> Option<Vec<u8>> {
229    use base64::prelude::*;
230    BASE64_STANDARD.decode(input).ok()
231}

233fn authenticator() -> MockLdapAuthenticator {
234    MockLdapAuthenticator
235}

237fn authorizer() -> RequestMatcherAuthorizer {
238    AuthorizationManager::request_matcher()
239        .http_basic()
240        .add_matcher("/admin.*", Access::new().roles(vec!["ROLE_ADMINS"]))
241        .add_matcher("/profile.*", Access::new().roles(vec!["ROLE_USERS"]))
242}

244#[actix_web::main]
245async fn main() -> std::io::Result<()> {
246    env_logger::init_from_env(env_logger::Env::default().default_filter_or("info"));
247
248    // Initialize the static state
249    let _ = get_mock_client();
250    let _ = get_ldap_config();
251
252    println!("=== LDAP Authentication Example ===");
253    println!("Server running at http://127.0.0.1:8080");
254    println!();
255    println!("Using MOCK LDAP with test users:");
256    println!("  john.doe:password   - Roles: [ROLE_USERS]");
257    println!("  jane.smith:secret   - Roles: [ROLE_USERS, ROLE_MANAGERS]");
258    println!("  admin:admin123      - Roles: [ROLE_USERS, ROLE_ADMINS]");
259    println!();
260    println!("Endpoints:");
261    println!("  GET /          - Home (requires auth)");
262    println!("  GET /admin     - Admin only (requires ROLE_ADMINS)");
263    println!("  GET /profile   - User profile (requires ROLE_USERS)");
264    println!("  GET /health    - Health check (public)");
265    println!();
266    println!("Try:");
267    println!("  curl -u john.doe:password http://127.0.0.1:8080/");
268    println!("  curl -u admin:admin123 http://127.0.0.1:8080/admin");
269    println!();
270
271    HttpServer::new(move || {
272        App::new()
273            .wrap(
274                SecurityTransform::new()
275                    .config_authenticator(authenticator)
276                    .config_authorizer(authorizer),
277            )
278            .service(index)
279            .service(admin)
280            .service(profile)
281            .service(health)
282    })
283    .bind("127.0.0.1:8080")?
284    .run()
285    .await
286}
