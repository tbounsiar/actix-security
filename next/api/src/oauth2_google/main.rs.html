1//! Google OAuth2 Authentication Example
2//!
3//! This example demonstrates OAuth2 authentication with Google.
4//!
5//! ## Prerequisites
6//!
7//! 1. Create a Google Cloud project at https://console.cloud.google.com
8//! 2. Enable the Google+ API or Google Identity API
9//! 3. Create OAuth2 credentials (Web application type)
10//! 4. Add authorized redirect URI: http://localhost:8080/auth/callback
11//!
12//! ## Running the example
13//!
14//! ```bash
15//! export GOOGLE_CLIENT_ID=your-client-id.apps.googleusercontent.com
16//! export GOOGLE_CLIENT_SECRET=your-client-secret
17//!
18//! cargo run -p actix-security-examples --bin oauth2_google
19//! ```
20//!
21//! ## Testing
22//!
23//! Open http://localhost:8080 in your browser and click "Login with Google"
24

32/// Application configuration
33#[derive(Clone)]
34struct AppConfig {
35    client_id: String,
36    client_secret: String,
37    redirect_uri: String,
38}
39
40impl AppConfig {
41    fn from_env() -> Self {
42        Self {
43            client_id: env::var("GOOGLE_CLIENT_ID")
44                .unwrap_or_else(|_| "your-client-id.apps.googleusercontent.com".to_string()),
45            client_secret: env::var("GOOGLE_CLIENT_SECRET")
46                .unwrap_or_else(|_| "your-client-secret".to_string()),
47            redirect_uri: env::var("REDIRECT_URI")
48                .unwrap_or_else(|_| "http://localhost:8080/auth/callback".to_string()),
49        }
50    }
51
52    fn is_configured(&self) -> bool {
53        !self.client_id.contains("your-client-id")
54    }
55}

198/// Start OAuth2 login flow
199#[get("/auth/login")]
200async fn login(session: Session, data: web::Data<Arc<AppState>>) -> impl Responder {
201    if !data.config.is_configured() {
202        return HttpResponse::BadRequest().body("Google OAuth2 not configured");
203    }
204
205    // Generate state for CSRF protection
206    let state = generate_state();
207    session.insert("oauth2_state", &state).ok();
208
209    // Build Google authorization URL
210    let auth_url = format!(
211        "https://accounts.google.com/o/oauth2/v2/auth?\
212        client_id={}&\
213        redirect_uri={}&\
214        response_type=code&\
215        scope=openid%20email%20profile&\
216        state={}&\
217        access_type=offline&\
218        prompt=consent",
219        urlencoding::encode(&data.config.client_id),
220        urlencoding::encode(&data.config.redirect_uri),
221        state
222    );
223
224    HttpResponse::Found()
225        .append_header(("Location", auth_url))
226        .finish()
227}

229/// OAuth2 callback - exchange code for tokens
230#[get("/auth/callback")]
231async fn callback(
232    session: Session,
233    data: web::Data<Arc<AppState>>,
234    query: web::Query<CallbackQuery>,
235) -> impl Responder {
236    // Check for error
237    if let Some(ref error) = query.error {
238        return HttpResponse::BadRequest().body(format!("OAuth2 error: {}", error));
239    }
240
241    let code = match &query.code {
242        Some(c) => c,
243        None => return HttpResponse::BadRequest().body("Missing authorization code"),
244    };
245
246    // Verify state
247    let stored_state: Option<String> = session.get("oauth2_state").ok().flatten();
248    if stored_state.as_ref() != query.state.as_ref() {
249        return HttpResponse::BadRequest().body("Invalid state parameter");
250    }
251    session.remove("oauth2_state");
252
253    // Exchange code for tokens
254    let client = reqwest::Client::new();
255    let token_response = client
256        .post("https://oauth2.googleapis.com/token")
257        .form(&[
258            ("client_id", data.config.client_id.as_str()),
259            ("client_secret", data.config.client_secret.as_str()),
260            ("code", code.as_str()),
261            ("redirect_uri", data.config.redirect_uri.as_str()),
262            ("grant_type", "authorization_code"),
263        ])
264        .send()
265        .await;
266
267    let tokens: TokenResponse = match token_response {
268        Ok(resp) => {
269            if !resp.status().is_success() {
270                let error_text = resp.text().await.unwrap_or_default();
271                return HttpResponse::InternalServerError()
272                    .body(format!("Token exchange failed: {}", error_text));
273            }
274            match resp.json().await {
275                Ok(t) => t,
276                Err(e) => {
277                    return HttpResponse::InternalServerError()
278                        .body(format!("Token parse error: {}", e))
279                }
280            }
281        }
282        Err(e) => {
283            return HttpResponse::InternalServerError().body(format!("Token exchange error: {}", e))
284        }
285    };
286
287    // Get user info
288    let userinfo_response = client
289        .get("https://www.googleapis.com/oauth2/v2/userinfo")
290        .bearer_auth(&tokens.access_token)
291        .send()
292        .await;
293
294    let userinfo: UserInfoResponse = match userinfo_response {
295        Ok(resp) => match resp.json().await {
296            Ok(u) => u,
297            Err(e) => {
298                return HttpResponse::InternalServerError()
299                    .body(format!("Userinfo parse error: {}", e))
300            }
301        },
302        Err(e) => {
303            return HttpResponse::InternalServerError().body(format!("Userinfo error: {}", e))
304        }
305    };
306
307    // Store user in session
308    let google_user = GoogleUser {
309        id: userinfo.id,
310        email: userinfo.email,
311        name: userinfo.name,
312        picture: userinfo.picture,
313        verified_email: userinfo.verified_email.unwrap_or(false),
314    };
315    session.insert("google_user", &google_user).ok();
316
317    // Redirect to home
318    HttpResponse::Found()
319        .append_header(("Location", "/"))
320        .finish()
321}

368/// Generate a random state string
369fn generate_state() -> String {
370    use std::time::{SystemTime, UNIX_EPOCH};
371    let timestamp = SystemTime::now()
372        .duration_since(UNIX_EPOCH)
373        .unwrap()
374        .as_nanos();
375    format!("{:032x}", timestamp)
376}

378#[actix_web::main]
379async fn main() -> std::io::Result<()> {
380    env_logger::init_from_env(env_logger::Env::default().default_filter_or("info"));
381
382    let config = AppConfig::from_env();
383
384    // Build OAuth2 config using actix-security
385    let oauth2_config = OAuth2Config::new(
386        &config.client_id,
387        &config.client_secret,
388        &config.redirect_uri,
389    )
390    .provider(OAuth2Provider::Google)
391    .scopes(vec!["openid", "email", "profile"]);
392
393    let state = Arc::new(AppState {
394        config: config.clone(),
395        oauth2_config,
396    });
397
398    let secret_key = Key::generate();
399
400    println!("=== Google OAuth2 Example ===");
401    println!("Server running at http://localhost:8080");
402    println!();
403
404    if config.is_configured() {
405        println!("Google OAuth2 configured:");
406        println!(
407            "  Client ID: {}...",
408            &config.client_id[..20.min(config.client_id.len())]
409        );
410        println!("  Redirect URI: {}", config.redirect_uri);
411    } else {
412        println!("⚠️  Google OAuth2 NOT configured!");
413        println!();
414        println!("To configure, set environment variables:");
415        println!("  export GOOGLE_CLIENT_ID=your-client-id.apps.googleusercontent.com");
416        println!("  export GOOGLE_CLIENT_SECRET=your-client-secret");
417        println!();
418        println!("Get credentials from: https://console.cloud.google.com/apis/credentials");
419    }
420
421    println!();
422    println!("Open http://localhost:8080 in your browser.");
423    println!();
424
425    HttpServer::new(move || {
426        App::new()
427            .wrap(
428                SessionMiddleware::builder(CookieSessionStore::default(), secret_key.clone())
429                    .cookie_secure(false)
430                    .build(),
431            )
432            .app_data(web::Data::new(state.clone()))
433            .service(index)
434            .service(login)
435            .service(callback)
436            .service(logout)
437    })
438    .bind("127.0.0.1:8080")?
439    .run()
440    .await
