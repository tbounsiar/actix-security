1//! OIDC Authentication with Keycloak Example
2//!
3//! This example demonstrates OpenID Connect authentication with Keycloak.
4//!
5//! # Prerequisites
6//!
7//! 1. Run Keycloak with Docker:
8//!    ```bash
9//!    docker run -p 8180:8080 -e KEYCLOAK_ADMIN=admin -e KEYCLOAK_ADMIN_PASSWORD=admin \
10//!      quay.io/keycloak/keycloak:latest start-dev
11//!    ```
12//!
13//! 2. Configure Keycloak:
14//!    - Create a realm: `actix-demo`
15//!    - Create a client: `actix-app` (confidential, with client secret)
16//!    - Set Valid Redirect URIs: `http://localhost:8080/*`
17//!    - Create a user with password
18//!
19//! # Run
20//! ```bash
21//! # Set environment variables
22//! export KEYCLOAK_URL=http://localhost:8180
23//! export KEYCLOAK_REALM=actix-demo
24//! export KEYCLOAK_CLIENT_ID=actix-app
25//! export KEYCLOAK_CLIENT_SECRET=your-client-secret
26//!
27//! cargo run --bin oidc_keycloak
28//! ```
29//!
30//! # Test
31//! Open http://localhost:8080 in a browser.
32//! Click "Login with Keycloak" to start the OIDC flow.

41/// Application configuration
42#[derive(Clone)]
43struct AppConfig {
44    keycloak_url: String,
45    realm: String,
46    client_id: String,
47    client_secret: String,
48    redirect_uri: String,
49}
50
51impl AppConfig {
52    fn from_env() -> Self {
53        Self {
54            keycloak_url: env::var("KEYCLOAK_URL")
55                .unwrap_or_else(|_| "http://localhost:8180".to_string()),
56            realm: env::var("KEYCLOAK_REALM").unwrap_or_else(|_| "actix-demo".to_string()),
57            client_id: env::var("KEYCLOAK_CLIENT_ID").unwrap_or_else(|_| "actix-app".to_string()),
58            client_secret: env::var("KEYCLOAK_CLIENT_SECRET")
59                .unwrap_or_else(|_| "change-me".to_string()),
60            redirect_uri: env::var("REDIRECT_URI")
61                .unwrap_or_else(|_| "http://localhost:8081/auth/callback".to_string()),
62        }
63    }
64
65    fn issuer_url(&self) -> String {
66        format!("{}/realms/{}", self.keycloak_url, self.realm)
67    }
68
69    fn authorization_endpoint(&self) -> String {
70        format!(
71            "{}/realms/{}/protocol/openid-connect/auth",
72            self.keycloak_url, self.realm
73        )
74    }
75
76    fn token_endpoint(&self) -> String {
77        format!(
78            "{}/realms/{}/protocol/openid-connect/token",
79            self.keycloak_url, self.realm
80        )
81    }
82
83    fn userinfo_endpoint(&self) -> String {
84        format!(
85            "{}/realms/{}/protocol/openid-connect/userinfo",
86            self.keycloak_url, self.realm
87        )
88    }
89
90    fn logout_endpoint(&self) -> String {
91        format!(
92            "{}/realms/{}/protocol/openid-connect/logout",
93            self.keycloak_url, self.realm
94        )
95    }
96}

186/// Start OAuth2/OIDC login flow
187#[get("/auth/login")]
188async fn login(session: Session, data: web::Data<Arc<AppState>>) -> impl Responder {
189    // Generate state for CSRF protection
190    let state = uuid_simple();
191    session.insert("oauth2_state", &state).ok();
192
193    // Build authorization URL
194    let auth_url = format!(
195        "{}?client_id={}&redirect_uri={}&response_type=code&scope=openid%20profile%20email&state={}",
196        data.config.authorization_endpoint(),
197        data.config.client_id,
198        urlencoding::encode(&data.config.redirect_uri),
199        state
200    );
201
202    HttpResponse::Found()
203        .append_header(("Location", auth_url))
204        .finish()
205}

207/// OAuth2 callback - exchange code for tokens
208#[get("/auth/callback")]
209async fn callback(
210    session: Session,
211    data: web::Data<Arc<AppState>>,
212    query: web::Query<CallbackQuery>,
213) -> impl Responder {
214    // Verify state
215    let stored_state: Option<String> = session.get("oauth2_state").ok().flatten();
216    if stored_state.as_ref() != Some(&query.state) {
217        return HttpResponse::BadRequest().body("Invalid state parameter");
218    }
219    session.remove("oauth2_state");
220
221    // Exchange code for tokens
222    let client = reqwest::Client::new();
223    let token_response = client
224        .post(data.config.token_endpoint())
225        .form(&[
226            ("grant_type", "authorization_code"),
227            ("client_id", &data.config.client_id),
228            ("client_secret", &data.config.client_secret),
229            ("code", &query.code),
230            ("redirect_uri", &data.config.redirect_uri),
231        ])
232        .send()
233        .await;
234
235    let tokens: TokenResponse = match token_response {
236        Ok(resp) => match resp.json().await {
237            Ok(t) => t,
238            Err(e) => {
239                return HttpResponse::InternalServerError()
240                    .body(format!("Token parse error: {}", e))
241            }
242        },
243        Err(e) => {
244            return HttpResponse::InternalServerError().body(format!("Token exchange error: {}", e))
245        }
246    };
247
248    // Get user info
249    let userinfo_response = client
250        .get(data.config.userinfo_endpoint())
251        .bearer_auth(&tokens.access_token)
252        .send()
253        .await;
254
255    let userinfo: UserInfoResponse = match userinfo_response {
256        Ok(resp) => match resp.json().await {
257            Ok(u) => u,
258            Err(e) => {
259                return HttpResponse::InternalServerError()
260                    .body(format!("Userinfo parse error: {}", e))
261            }
262        },
263        Err(e) => {
264            return HttpResponse::InternalServerError().body(format!("Userinfo error: {}", e))
265        }
266    };
267
268    // Extract roles from Keycloak token (realm_access.roles or resource_access)
269    let roles = extract_roles_from_token(&tokens.access_token);
270
271    // Store user in session
272    let session_user = SessionUser {
273        username: userinfo
274            .preferred_username
275            .unwrap_or_else(|| userinfo.sub.clone()),
276        email: userinfo.email,
277        name: userinfo.name,
278        roles,
279    };
280    session.insert("user", &session_user).ok();
281
282    // Redirect to home
283    HttpResponse::Found()
284        .append_header(("Location", "/"))
285        .finish()
286}

288/// Logout
289#[get("/logout")]
290async fn logout(session: Session, data: web::Data<Arc<AppState>>) -> impl Responder {
291    session.purge();
292
293    // Redirect to Keycloak logout
294    let logout_url = format!(
295        "{}?redirect_uri={}",
296        data.config.logout_endpoint(),
297        urlencoding::encode("http://localhost:8081/")

299
300    HttpResponse::Found()
301        .append_header(("Location", logout_url))
302        .finish()
303}

340/// Simple UUID generator (for state parameter)
341fn uuid_simple() -> String {
342    use std::time::{SystemTime, UNIX_EPOCH};
343    let timestamp = SystemTime::now()
344        .duration_since(UNIX_EPOCH)
345        .unwrap()
346        .as_nanos();
347    format!("{:032x}", timestamp)
348}

350/// Extract roles from JWT access token (Keycloak format)
351fn extract_roles_from_token(token: &str) -> Vec<String> {
352    // Decode JWT payload (second part)
353    let parts: Vec<&str> = token.split('.').collect();
354    if parts.len() != 3 {
355        return vec![];
356    }
357
358    // Decode base64url
359    let payload = match base64_url_decode(parts[1]) {
360        Some(p) => p,
361        None => return vec![],
362    };
363
364    // Parse JSON
365    let claims: serde_json::Value = match serde_json::from_slice(&payload) {
366        Ok(c) => c,
367        Err(_) => return vec![],
368    };
369
370    // Extract roles from realm_access.roles
371    let mut roles = vec![];
372    if let Some(realm_access) = claims.get("realm_access") {
373        if let Some(realm_roles) = realm_access.get("roles").and_then(|r| r.as_array()) {
374            for role in realm_roles {
375                if let Some(r) = role.as_str() {
376                    roles.push(r.to_string());
377                }
378            }
379        }
380    }
381
382    roles
383}

385fn base64_url_decode(input: &str) -> Option<Vec<u8>> {
386    use base64::prelude::*;
387
388    // Add padding if needed
389    let padded = match input.len() % 4 {
390        2 => format!("{}==", input),
391        3 => format!("{}=", input),
392        _ => input.to_string(),
393    };
394
395    // Replace URL-safe characters
396    let standard = padded.replace('-', "+").replace('_', "/");
397
398    BASE64_STANDARD.decode(&standard).ok()
399}

401#[actix_web::main]
402async fn main() -> std::io::Result<()> {
403    env_logger::init_from_env(env_logger::Env::default().default_filter_or("info"));
404
405    let config = AppConfig::from_env();
406
407    // Build OAuth2 config (for reference, we do manual flow here for clarity)
408    let oauth2_config = OAuth2Config::new(
409        &config.client_id,
410        &config.client_secret,
411        &config.redirect_uri,
412    )
413    .authorization_uri(config.authorization_endpoint())
414    .token_uri(config.token_endpoint())
415    .userinfo_uri(config.userinfo_endpoint())
416    .scopes(vec!["openid", "profile", "email"]);
417
418    let state = Arc::new(AppState {
419        config: config.clone(),
420        oauth2_config,
421    });
422
423    let secret_key = Key::generate();
424
425    println!("=== OIDC Keycloak Example ===");
426    println!("Server running at http://localhost:8081");
427    println!();
428    println!("Keycloak configuration:");
429    println!("  URL: {}", config.keycloak_url);
430    println!("  Realm: {}", config.realm);
431    println!("  Client ID: {}", config.client_id);
432    println!("  Issuer: {}", config.issuer_url());
433    println!();
434    println!("Make sure Keycloak is running and configured:");
435    println!(
436        "  docker run -p 8180:8080 -e KEYCLOAK_ADMIN=admin -e KEYCLOAK_ADMIN_PASSWORD=admin \\"
437    );
438    println!("    quay.io/keycloak/keycloak:latest start-dev");
439    println!();
440    println!("Open http://localhost:8081 in your browser.");
441    println!();
442
443    HttpServer::new(move || {
444        App::new()
445            .wrap(
446                SessionMiddleware::builder(CookieSessionStore::default(), secret_key.clone())
447                    .cookie_secure(false)
448                    .build(),
449            )
450            .app_data(web::Data::new(state.clone()))
451            .service(index)
452            .service(login)
453            .service(callback)
454            .service(logout)
455    })
456    .bind("127.0.0.1:8081")?
457    .run()
458    .await
