1//! SAML 2.0 Authentication Example
2//!
3//! This example demonstrates SAML 2.0 Single Sign-On (SSO) authentication.
4//! It implements the Service Provider (SP) side of SAML authentication.
5//!
6//! ## Running the example
7//!
8//! ```bash
9//! cargo run -p actix-security-examples --bin saml_auth
10//! ```
11//!
12//! ## SAML Flow Overview
13//!
14//! 1. User accesses protected resource on SP
15//! 2. SP generates AuthnRequest and redirects to IdP (Identity Provider)
16//! 3. User authenticates at IdP
17//! 4. IdP sends SAML Response back to SP's ACS (Assertion Consumer Service) URL
18//! 5. SP validates response and creates session
19//!
20//! ## Testing with a SAML IdP
21//!
22//! For testing, you can use:
23//! - Keycloak (supports SAML 2.0)
24//! - SimpleSAMLphp
25//! - Okta Developer
26//! - Auth0
27//!
28//! ## Configuration
29//!
30//! Set these environment variables:
31//! ```bash
32//! export SAML_ENTITY_ID=https://myapp.example.com/saml/metadata
33//! export SAML_IDP_SSO_URL=https://idp.example.com/saml/sso
34//! export SAML_ACS_URL=http://localhost:8080/saml/acs
35//! export SAML_IDP_CERTIFICATE=/path/to/idp-cert.pem
36//! ```

240/// Assertion Consumer Service (receives SAML Response from IdP)
241#[post("/saml/acs")]
242async fn acs(
243    session: Session,
244    data: web::Data<Arc<AppState>>,
245    form: web::Form<AcsForm>,
246) -> impl Responder {
247    println!("SAML Response received at ACS");
248
249    // Process and validate the SAML response
250    match data
251        .saml_authenticator
252        .process_response(&form.saml_response)
253    {
254        Ok(auth_result) => {
255            // Extract user information
256            let saml_user = SamlUser {
257                name_id: auth_result.name_id.clone(),
258                session_index: auth_result.session_index.clone(),
259                attributes: auth_result.attributes.clone(),
260            };
261
262            // Store in session
263            session.insert("saml_user", &saml_user).ok();
264
265            println!("SAML authentication successful for: {}", saml_user.name_id);
266
267            // Redirect to home
268            HttpResponse::Found()
269                .append_header(("Location", "/"))
270                .finish()
271        }
272        Err(e) => {
273            println!("SAML validation error: {:?}", e);
274            HttpResponse::BadRequest().body(format!("SAML validation failed: {:?}", e))
275        }
276    }
277}

342fn create_saml_config() -> SamlConfig {
343    // Load from environment or use defaults
344    let entity_id = env::var("SAML_ENTITY_ID")
345        .unwrap_or_else(|_| "http://localhost:8080/saml/metadata".to_string());
346    let idp_sso_url = env::var("SAML_IDP_SSO_URL")
347        .unwrap_or_else(|_| "https://idp.example.com/saml/sso".to_string());
348    let acs_url =
349        env::var("SAML_ACS_URL").unwrap_or_else(|_| "http://localhost:8080/saml/acs".to_string());
350
351    SamlConfig::new()
352        .entity_id(&entity_id)
353        .idp_sso_url(&idp_sso_url)
354        .assertion_consumer_service_url(&acs_url)
355        .name_id_format(NameIdFormat::EmailAddress)
356        .authn_context_class(AuthnContextClass::PasswordProtectedTransport)
357        .sso_binding(SamlBinding::HttpRedirect)
358        .sign_authn_request(false) // Set to true in production with proper certificates
359        .want_assertions_signed(true)
360}

362#[actix_web::main]
363async fn main() -> std::io::Result<()> {
364    env_logger::init_from_env(env_logger::Env::default().default_filter_or("info"));
365
366    let saml_config = create_saml_config();
367    let saml_authenticator =
368        SamlAuthenticator::new(saml_config.clone()).expect("Failed to create SAML authenticator");
369
370    let state = Arc::new(AppState {
371        saml_config: saml_config.clone(),
372        saml_authenticator,
373    });
374
375    let secret_key = Key::generate();
376
377    println!("=== SAML 2.0 Authentication Example ===");
378    println!("Server running at http://127.0.0.1:8080");
379    println!();
380    println!("SAML Configuration:");
381    println!("  Entity ID (SP): {}", saml_config.entity_id);
382    println!("  IdP SSO URL: {}", saml_config.idp_sso_url);
383    println!("  ACS URL: {}", saml_config.acs_url);
384    println!();
385    println!("Endpoints:");
386    println!("  GET  /                  - Home page");
387    println!("  GET  /saml/metadata     - SP Metadata (XML)");
388    println!("  GET  /saml/login        - Initiate SAML SSO");
389    println!("  POST /saml/acs          - Assertion Consumer Service");
390    println!("  POST /saml/logout       - Logout");
391    println!("  GET  /saml/demo-response - Simulate SAML response (for testing)");
392    println!();
393    println!("For testing without a real IdP, visit:");
394    println!("  http://127.0.0.1:8080/saml/demo-response");
395    println!();
396
397    HttpServer::new(move || {
398        App::new()
399            .wrap(
400                SessionMiddleware::builder(CookieSessionStore::default(), secret_key.clone())
401                    .cookie_secure(false)
402                    .build(),
403            )
404            .app_data(web::Data::new(state.clone()))
405            .service(index)
406            .service(metadata)
407            .service(saml_login)
408            .service(acs)
409            .service(demo_response)
410            .service(saml_logout)
411    })
412    .bind("127.0.0.1:8080")?
413    .run()
414    .await
