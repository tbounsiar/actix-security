1//! Session-based Authentication Example
2//!
3//! This example demonstrates session-based authentication with session fixation protection.
4//!
5//! # Run
6//! ```bash
7//! cargo run --example session_auth
8//! ```
9//!
10//! # Test
11//! ```bash
12//! # Login (stores session cookie)
13//! curl -c cookies.txt -X POST http://localhost:8080/login \
14//!   -H "Content-Type: application/json" \
15//!   -d '{"username":"admin","password":"admin"}'
16//!
17//! # Access protected route with session
18//! curl -b cookies.txt http://localhost:8080/dashboard
19//!
20//! # Logout
21//! curl -b cookies.txt -c cookies.txt -X POST http://localhost:8080/logout
22//! ```
23
24use actix_security::http::security::{
25    Argon2PasswordEncoder, PasswordEncoder, SessionAuthenticator, SessionConfig,
26    SessionFixationStrategy, User,
27};
28use actix_session::{storage::CookieSessionStore, Session, SessionMiddleware};
29use actix_web::{cookie::Key, get, post, web, App, HttpResponse, HttpServer, Responder};
30use serde::Deserialize;
31use std::sync::Arc;
32
33/// Login credentials
34#[derive(Deserialize)]
35struct LoginRequest {
36    username: String,
37    password: String,
38}
39
40/// Application state
41struct AppState {
42    users: Vec<User>,
43    encoder: Argon2PasswordEncoder,
44    session_config: SessionConfig,
45}
46
47/// Login endpoint
48#[post("/login")]
49async fn login(
50    session: Session,
51    data: web::Data<Arc<AppState>>,
52    credentials: web::Json<LoginRequest>,
53) -> impl Responder {
54    // Find user and verify password
55    let user = data
56        .users
57        .iter()
58        .find(|u| u.get_username() == credentials.username);
59
60    match user {
61        Some(user)
62            if data
63                .encoder
64                .matches(&credentials.password, user.get_password()) =>
65        {
66            // Login with session fixation protection
67            match SessionAuthenticator::login(&session, user, &data.session_config) {
68                Ok(()) => HttpResponse::Ok().json(serde_json::json!({
69                    "message": "Login successful",
70                    "username": user.get_username(),
71                    "roles": user.get_roles()
72                })),
73                Err(e) => HttpResponse::InternalServerError().body(format!("Session error: {}", e)),
74            }
75        }
76        _ => HttpResponse::Unauthorized().json(serde_json::json!({
77            "error": "Invalid credentials"
78        })),
79    }
80}
81
82/// Logout endpoint
83#[post("/logout")]
84async fn logout(session: Session, data: web::Data<Arc<AppState>>) -> impl Responder {
85    SessionAuthenticator::logout(&session, &data.session_config);
86    HttpResponse::Ok().json(serde_json::json!({
87        "message": "Logged out successfully"
88    }))
89}

91/// Protected dashboard
92#[get("/dashboard")]
93async fn dashboard(session: Session, data: web::Data<Arc<AppState>>) -> impl Responder {
94    // Check if user is authenticated via session
95    match SessionAuthenticator::get_session_user(&session, &data.session_config) {
96        Some(user) => HttpResponse::Ok().json(serde_json::json!({
97            "message": format!("Welcome to dashboard, {}!", user.get_username()),
98            "user": {
99                "username": user.get_username(),
100                "roles": user.get_roles(),
101                "authorities": user.get_authorities()
102            }
103        })),
104        None => HttpResponse::Unauthorized().json(serde_json::json!({
105            "error": "Not authenticated",
106            "hint": "Please login first at POST /login"
107        })),
108    }
109}

111/// Public home page
112#[get("/")]
113async fn index() -> impl Responder {
114    HttpResponse::Ok().json(serde_json::json!({
115        "message": "Session Authentication Example",
116        "endpoints": {
117            "POST /login": "Login with credentials",
118            "POST /logout": "Logout and clear session",
119            "GET /dashboard": "Protected dashboard (requires session)"
120        }
121    }))
122}

124#[actix_web::main]
125async fn main() -> std::io::Result<()> {
126    env_logger::init_from_env(env_logger::Env::default().default_filter_or("info"));
127
128    let encoder = Argon2PasswordEncoder::new();
129
130    // Create test users
131    let users = vec![
132        User::with_encoded_password("admin", encoder.encode("admin"))
133            .roles(&["ADMIN".into(), "USER".into()]),
134        User::with_encoded_password("user", encoder.encode("user")).roles(&["USER".into()]),
135    ];
136
137    // Session configuration with fixation protection
138    let session_config = SessionConfig::new()
139        .user_key("user") // Key to store user in session
140        .fixation_strategy(SessionFixationStrategy::MigrateSession); // Protect against session fixation
141
142    let state = Arc::new(AppState {
143        users,
144        encoder,
145        session_config,
146    });
147
148    // Generate a random key for session encryption
149    let secret_key = Key::generate();
150
151    println!("=== Session Authentication Example ===");
152    println!("Server running at http://localhost:8080");
153    println!();
154    println!("Session features:");
155    println!("  - Session fixation protection (MigrateSession)");
156    println!("  - 30-minute session timeout");
157    println!("  - Secure cookie-based sessions");
158    println!();
159    println!("Test users: admin:admin, user:user");
160    println!();
161    println!("Try:");
162    println!("  # Login");
163    println!("  curl -c cookies.txt -X POST http://localhost:8080/login \\");
164    println!("    -H 'Content-Type: application/json' \\");
165    println!("    -d '{{\"username\":\"admin\",\"password\":\"admin\"}}'");
166    println!();
167    println!("  # Access dashboard");
168    println!("  curl -b cookies.txt http://localhost:8080/dashboard");
169    println!();
170
171    HttpServer::new(move || {
172        App::new()
173            .wrap(
174                SessionMiddleware::builder(CookieSessionStore::default(), secret_key.clone())
175                    .cookie_secure(false) // Set to true in production with HTTPS
176                    .build(),
177            )
178            .app_data(web::Data::new(state.clone()))
179            .service(index)
180            .service(login)
181            .service(logout)
182            .service(dashboard)
183    })
184    .bind("127.0.0.1:8080")?
185    .run()
186    .await
187}
